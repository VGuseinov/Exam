задача 1

delimiter $$

create function func1 (employeeid int, tipsDate datetime, tipsAmount int)
return float
deterministic
begin
    declare avgTips float;

    insert into tips(employee_id, date, amount) values(employeeid, tipsDate, tipsAmount);

    select sum(amount) into avgTips from tips where employee_id = employeeid;

    return avgTips

end $$
